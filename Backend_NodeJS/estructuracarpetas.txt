comando para validar estructuras:

Get-ChildItem -Recurse | Where-Object { $_.FullName -notmatch "node_modules" } | ForEach-Object {
    $depth = ($_.FullName.Replace((Get-Location).Path, "").Split('\').Count - 1)
    $prefix = if ($_.PSIsContainer) { "[DIR] " } else { "|-- " }
    "  " * $depth + $prefix + $_.Name
}
